# share-tmux.sh
#!/bin/bash
SH_PATH="$(dirname $0)/"
SESSION="${SH_PATH}.sessions/`whoami`-session"
#TERM=dtterm
${SH_PATH}tmux -f ${SH_PATH}tmux.conf -S $SESSION new -d -s `whoami`
chmod 644 $SESSION
${SH_PATH}tmux -S $SESSION attach -t `whoami`
rm $SESSION

# attach-tmux.sh
#!/bin/bash
if [ $# != 1 ];then
    echo "give me user you want to attach"
    exit
fi
SH_PATH="$(dirname $0)/"
SESSION="${SH_PATH}.sessions/$1-session"
#TERM=dtterm
${SH_PATH}tmux -S $SESSION attach -t $1
